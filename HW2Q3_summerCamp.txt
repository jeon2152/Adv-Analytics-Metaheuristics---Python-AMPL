#AMPL model for "Summer Camp" Problem (Q3(b) of Assignment #2)
# This solves the problem for an instance of 6 persons

reset;

#options
options solver cplex;

#sets 
set I = {1 .. 6};  
set J = {1 .. 6};  

#parameters
param b{I,J};

#decision variables
var x{I,J} >= 0;  

#data
data Q3_summerCamp.dat;

#objective
maximize cost: sum {i in I, j in J} b[i,j] * x[i,j];


#constraints

subject to pairness {i in I}: sum {j in J} x[i,j] = 1;
subject to pairness2 {j in J}: sum {i in I} x[i,j] = 1;

#commands

solve;
display {i in I, j in J} x[i,j];
